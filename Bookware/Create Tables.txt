


CREATE TABLE BwUnderviser( Underviser_Id int NOT NULL PRIMARY KEY IDENTITY(1,1), Name varchar(50) NOT NULL);
CREATE TABLE BwFag( Fag_Id int NOT NULL PRIMARY KEY IDENTITY(1,1), FagName varchar(50) NOT NULL);

CREATE TABLE BwUnderviser_Fag( UndF_Id int IDENTITY(1,1) PRIMARY KEY, Underviser_Id int NOT NULL, Fag_Id int NOT NULL, 
FOREIGN KEY (Underviser_Id) REFERENCES BwUnderviser(Underviser_Id) ON UPDATE NO ACTION ON DELETE CASCADE, 
FOREIGN KEY (Fag_Id) REFERENCES BwFag(Fag_Id) ON UPDATE NO ACTION ON DELETE CASCADE);

CREATE TABLE BwUddannelse( Udd_Id int NOT NULL PRIMARY KEY IDENTITY(1,1), UddName varchar(50) NOT NULL);

CREATE TABLE BwUddannelses_Fag( UddF_Id int IDENTITY(1,1) PRIMARY KEY, Udd_Id int NOT NULL, Fag_Id int NOT NULL, 
FOREIGN KEY (Udd_Id) REFERENCES BwUddannelse(Udd_Id) ON UPDATE NO ACTION ON DELETE CASCADE, 
FOREIGN KEY (Fag_Id) REFERENCES BwFag(Fag_Id) ON UPDATE NO ACTION ON DELETE CASCADE);

CREATE TABLE BwStuderende( Student_Id int NOT NULL PRIMARY KEY IDENTITY(1,1), StudentName varchar(50) NOT NULL, Adress varchar(70) NOT NULL);

CREATE TABLE BwHold( Hold_Id int IDENTITY(1,1) PRIMARY KEY, HoldName varchar(50) NOT NULL, Semester int NOT NULL, 
UddF_Id int NOT NULL, UndF_Id int NOT NULL, Student_Id int NOT NULL
FOREIGN KEY (UddF_Id) REFERENCES BwUddannelses_Fag(UddF_Id), FOREIGN KEY (UndF_Id) REFERENCES BwUnderviser_Fag(UndF_Id), 
FOREIGN KEY (Student_Id) REFERENCES BwStuderende(Student_Id));

CREATE TABLE BwHold_Bog (HB_Id INT NOT NULL PRIMARY KEY IDENTITY(1,1), Hold_Id INT NOT NULL, Bog_Id INT NOT NULL, 
FOREIGN KEY (Hold_Id) REFERENCES BwHold(Hold_Id) ON UPDATE NO ACTION ON DELETE CASCADE, 
FOREIGN KEY (Bog_Id) REFERENCES BwBog(Bog_Id) ON UPDATE NO ACTION ON DELETE CASCADE );

CREATE TABLE BwBog (Bog_Id INT NOT NULL PRIMARY KEY IDENTITY(1,1), Titel VARCHAR(50) NOT NULL, 
Author VARCHAR(50) NOT NULL, Year INT NOT NULL, ISBN INT NOT NULL, Amount INT NOT NULL);
